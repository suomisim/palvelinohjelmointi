CREATE TABLE webuser (id SMALLINT NOT NULL AUTO_INCREMENT PRIMARY KEY, username VARCHAR(255) NOT NULL UNIQUE KEY, password_hash VARCHAR(255) NOT NULL, salt VARCHAR(255) NOT NULL) ENGINE=InnoDB DEFAULT CHARSET=latin1;

CREATE TABLE webuser_weigth (id SMALLINT NOT NULL AUTO_INCREMENT PRIMARY KEY, userid SMALLINT NOT NULL, weigth NUMERIC (5,2) NOT NULL) ENGINE=InnoDB DEFAULT CHARSET=latin1;
