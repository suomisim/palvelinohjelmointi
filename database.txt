CREATE TABLE webuser_weigth (id SMALLINT NOT NULL AUTO_INCREMENT PRIMARY KEY, date VARCHAR(255) NOT NULL, userid SMALLINT NOT NULL, weigth NUMERIC (5,2) NOT NULL) ENGINE=InnoDB DEFAULT CHARSET=latin1;
